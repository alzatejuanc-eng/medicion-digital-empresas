¿Qué se guarda?

Al finalizar la evaluación, se envía a Google Sheets: empresa, evaluador, contacto, los 40 puntajes individuales por criterio, score total (40–200), nivel MIEDE 2.0, scores por pilar y el roadmap de recomendaciones generado.

1

Crear un Google Sheet nuevo

Ve a sheets.google.com → "+" hoja en blanco. Nómbrala "MIEDE 2.0 Evaluaciones".

2

Crear el Apps Script

En tu Sheet: Extensiones → Apps Script. Borra todo y pega este código:

function doPost(e) { try { var d = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // ── Hoja 1: Resumen ───────────────────────────── var h1 = ss.getSheetByName('Evaluaciones') || ss.insertSheet('Evaluaciones'); if (h1.getLastRow() === 0) { h1.appendRow([ 'ID','Fecha','Hora','Empresa','RUC','Sector','Tamaño', 'País','Ciudad','Web','Años Op.', 'Evaluador','Cargo','Email','Teléfono','Área','Exp.TD', 'Objetivo','Perspectiva','Descripción','Contexto', 'Score (40-200)','Nivel','Nombre Nivel','Subtítulo', // 8 pilares MIEDE 2.0 'P1-Evaluación IA','P2-Estrategia Digital', 'P3-Arquitectura IA','P4-Optimización', 'P5-Innovación GenAI','P6-Capacidades ExO', 'P7-Disrupción','P8-Fusión Digital' ]); } h1.appendRow([ d.id, d.fecha, d.hora, d.empresa, d.ruc, d.sector, d.tamano, d.pais, d.ciudad, d.web, d.anos, d.nombre, d.cargo, d.email, d.tel, d.area, d.exp, d.objetivo, d.perspectiva, d.descripcion, d.contexto, d.score, d.nivel, d.nivelNombre, d.subtitulo ].concat(d.pilScores)); // ── Hoja 2: Criterios detallados ──────────────── var h2 = ss.getSheetByName('Criterios Detalle') || ss.insertSheet('Criterios Detalle'); if (h2.getLastRow() === 0) { h2.appendRow([ 'EvalID','Empresa','Fecha', 'Pilar Nro','Pilar Nombre', 'ID Criterio','Pregunta', 'Score (1-5)','Nivel Texto','Observación' ]); } d.detalle.forEach(function(r) { h2.appendRow([ d.id, d.empresa, d.fecha, r.pilNum, r.pilName, r.id, r.txt, r.score, r.nivel, r.obs ]); }); return ok('OK'); } catch(err) { return ok('Error: ' + err.toString()); } } function ok(m) { return ContentService .createTextOutput(JSON.stringify({status:'ok',msg:m})) .setMimeType(ContentService.MimeType.JSON); }
